HtmlTagBuilder

class HtmlTagBuilder(name: String, renderer: BaseHtmlNodeRenderer, pretty: Boolean) : TagBuilder

A builder of an HTML tag.

Example:

val builder = tagBuilder("html") {
    tag("head") {
        tag("meta")
            .attribute("charset", "UTF-8")
            .void(true)
    }
    tag("body") {
        +node.children
    }
